Calculatrice de base

Saisissez deux nombres pour afficher la somme, la différence, le produit et le quotient.

Le résultat apparaîtra ici…
Script JS 
function calculer() {
  const a = parseFloat(document.getElementById("ex1-a").value);
  const b = parseFloat(document.getElementById("ex1-b").value);

  const somme      = a + b;
  const difference = a - b;
  const produit    = a * b;
  const quotient   = b !== 0 ? a / b : "indéfini (division par zéro)";

  console.log(`Somme      : ${somme}`);
  console.log(`Différence : ${difference}`);
  console.log(`Produit    : ${produit}`);
  console.log(`Quotient   : ${quotient}`);
}

Deviner le nombre

L'ordinateur choisit un nombre entre 1 et 10. Trouvez-le en le moins de tentatives possible !

let secretEx2, tentativesEx2, gagnéEx2;

function nouvellePartie() {

  tentativesEx2 = 0;
  gagnéEx2     = false;
  // réinitialise l'interface…
}

function proposer() {
  const guess = parseInt(document.getElementById("ex2-guess").value);
  tentativesEx2++;

  if (guess === secretEx2) {
    alert(`Bravo ! Trouvé en ${tentativesEx2} tentative(s).`);
  } else if (guess < secretEx2) {
    alert("Le nombre est plus grand !");
  } else {
    alert("Le nombre est plus petit !");
  }
}

const QUESTIONS = [
  ["Quelle est la capitale de la France ?", "Paris"],
  ["Combien de côtés a un hexagone ?", "6"],
  ["Quel est le plus grand océan ?", "Pacifique"],
  ["Qui a écrit Hamlet ?", "Shakespeare"],
  ["En quelle année a eu lieu la Révolution française ?", "1789"],
];

function lancerQuiz() {
  let score = 0;
  for (let i = 0; i < QUESTIONS.length; i++) {
    const [question, bonne] = QUESTIONS[i];
    const reponse = prompt(question);
    if (reponse !== null &&
        reponse.trim().toLowerCase() === bonne.toLowerCase()) {
      alert("Réponse juste !");
      score++;
    } else {
      alert("Réponse fausse.");
    }
  }
  alert(
    `Vous avez répondu correctement à ${score} questions sur ${QUESTIONS.length}.`
  );
}
